128 float DistancePointToSegment(glm::vec2 p, glm::vec2 a, glm::vec2 b) {
129 glm::vec2 ab = b - a;
130 glm::vec2 ap = p - a;
131 float t = glm::dot(ap, ab) / glm::dot(ab, ab);
132 t = glm::clamp(t, 0.0f, 1.0f);
133 glm::vec2 closest = a + t * ab;
134 return glm::length(p - closest);
135 }
